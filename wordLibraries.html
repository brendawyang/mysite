Stack 1, 2, and 3 grams together in df

# adds all the sums for 1, 2, and 3 grams. This results in a numerical variable called `autoMindsetNum`
df <- full_join(df, df2) %>%
  full_join(df3) %>%
  replace_na(list(growthMinusFixed = 0, growthMinusFixed2 = 0, growthMinusFixed3 = 0)) %>%
  mutate(autoMindsetNum = growthMinusFixed + growthMinusFixed2 + growthMinusFixed3) %>%
  select(researchID, autoMindsetNum, everything())
